 448def handle_feedback(request):
 449    """
 450    POST /api/feedback/
 451
 452    Trim a video based on provided start/end times and save feedback.
 453
 454    JSON Body:
 455        {
 456            "user_id": str,
 457            "feedback_text": str,
 458            "video_url": str,     # must include '/media/'
 459            "start_time": "HH:MM:SS",
 460            "end_time": "HH:MM:SS"
 461        }
 462
 463    Returns:
 464        200: {'status': 'success', 'id': uuid, 'trimmed_video': path}
 465        400: missing/invalid fields
 466        404: file not found
 467        500: processing or server error
 468    """
 469    if request.method != 'POST':
 470        return JsonResponse({'error': 'POST only'}, status=405)
 471
 472    try:
 473        data = json.loads(request.body)
 474        # Validate and parse times
 475        def valid_time(t):
 476            try:
 477                h, m, s = map(int, t.split(':'))
 478                return 0 <= h < 24 and 0 <= m < 60 and 0 <= s < 60
 479            except:
 480                return False
 481
 482        for field in ('start_time', 'end_time', 'user_id', 'feedback_text', 'video_url'):
 483            if field not in data:
 484                return JsonResponse({'error': f"Missing field '{field}'"}, status=400)
 485
 486        if not valid_time(data['start_time']) or not valid_time(data['end_time']):
 487            return JsonResponse({'error': 'Invalid time format'}, status=400)
 488
 489        # Convert HH:MM:SS to seconds
 490        def to_sec(t): return sum(int(x) * 60 ** i for i, x in enumerate(reversed(t.split(':'))))
 491
 492        start_sec, end_sec = to_sec(data['start_time']), to_sec(data['end_time'])
 493        if start_sec == end_sec:
 494            return JsonResponse({'error': 'Start and end cannot match'}, status=400)
 495        if start_sec > end_sec:
 496            start_sec, end_sec = end_sec, start_sec
 497
 498        duration = end_sec - start_sec
 499
 500        # Extract relative path from URL
 501        if '/media/' not in data['video_url']:
 502            return JsonResponse({'error': 'Invalid video path'}, status=400)
 503        rel = data['video_url'].split('/media/', 1)[1]
 504        input_path = Path(settings.BASE_DIR) / 'media' / rel
 505        if not input_path.exists():
 506            return JsonResponse({'error': f'Video not found: {input_path}'}, status=404)
 507
 508        # Prepare output
 509        uid = str(uuid.uuid4())
 510        out_dir = Path(settings.BASE_DIR) / 'feedbackvideos'
 511        out_dir.mkdir(parents=True, exist_ok=True)
 512        out_file = out_dir / f"{uid}.mp4"
 513
 514        # Run ffmpeg trim
 515        cmd = [
 516            'ffmpeg', '-ss', data['start_time'], '-i', str(input_path),
 517            '-t', str(duration), '-c:v', 'libx264', '-preset', 'ultrafast',
 518            '-an', str(out_file), '-y'
 519        ]
 520        res = subprocess.run(cmd, capture_output=True, text=True)
 521        if res.returncode != 0:
 522            return JsonResponse({'error': 'Processing failed', 'detail': res.stderr}, status=500)
 523
 524        # Save feedback record
 525        fb = Feedback.objects.create(
 526            unique_id=uid, user_id=data['user_id'],
 527            feedback_text=data['feedback_text'],
 528            video_url=rel,
 529            trimmed_video=f"feedbackvideos/{out_file.name}",
 530            start_time=data['start_time'],
 531            end_time=data['end_time']
 532        )
 533        return JsonResponse({'status': 'success', 'id': uid, 'trimmed_video': fb.trimmed_video})
 534
 535    except json.JSONDecodeError:
 536        return JsonResponse({'error': 'Invalid JSON'}, status=400)
 537    except Exception as e:
 538        return JsonResponse({'error': f'Server error: {e}'}, status=500)

450    """
451    POST /api/feedback/
452
453    Trim a video based on provided start/end times and save feedback.
454
455    JSON Body:
456        {
457            "user_id": str,
458            "feedback_text": str,
459            "video_url": str,     # must include '/media/'
460            "start_time": "HH:MM:SS",
461            "end_time": "HH:MM:SS"
462        }
463
464    Returns:
465        200: {'status': 'success', 'id': uuid, 'trimmed_video': path}
466        400: missing/invalid fields
467        404: file not found
468        500: processing or server error
469    """
470    if request.method != 'POST':
471        return JsonResponse({'error': 'POST only'}, status=405)
472
473    try:
474        data = json.loads(request.body)
475        # Validate and parse times
476        def valid_time(t):
477            try:
478                h, m, s = map(int, t.split(':'))
479                return 0 <= h < 24 and 0 <= m < 60 and 0 <= s < 60
480            except:
481                return False
482
483        for field in ('start_time', 'end_time', 'user_id', 'feedback_text', 'video_url'):
484            if field not in data:
485                return JsonResponse({'error': f"Missing field '{field}'"}, status=400)
486
487        if not valid_time(data['start_time']) or not valid_time(data['end_time']):
488            return JsonResponse({'error': 'Invalid time format'}, status=400)
489
490        # Convert HH:MM:SS to seconds
491        def to_sec(t): return sum(int(x) * 60 ** i for i, x in enumerate(reversed(t.split(':'))))
492
493        start_sec, end_sec = to_sec(data['start_time']), to_sec(data['end_time'])
494        if start_sec == end_sec:
495            return JsonResponse({'error': 'Start and end cannot match'}, status=400)
496        if start_sec > end_sec:
497            start_sec, end_sec = end_sec, start_sec
498
499        duration = end_sec - start_sec
500
501        # Extract relative path from URL
502        if '/media/' not in data['video_url']:
503            return JsonResponse({'error': 'Invalid video path'}, status=400)
504        rel = data['video_url'].split('/media/', 1)[1]
505        input_path = Path(settings.BASE_DIR) / 'media' / rel
506        if not input_path.exists():
507            return JsonResponse({'error': f'Video not found: {input_path}'}, status=404)
508
509        # Prepare output
510        uid = str(uuid.uuid4())
511        out_dir = Path(settings.BASE_DIR) / 'feedbackvideos'
512        out_dir.mkdir(parents=True, exist_ok=True)
513        out_file = out_dir / f"{uid}.mp4"
514
515        # Run ffmpeg trim
516        cmd = [
517            'ffmpeg', '-ss', data['start_time'], '-i', str(input_path),
518            '-t', str(duration), '-c:v', 'libx264', '-preset', 'ultrafast',
519            '-an', str(out_file), '-y'
520        ]
521        res = subprocess.run(cmd, capture_output=True, text=True)
522        if res.returncode != 0:
523            return JsonResponse({'error': 'Processing failed', 'detail': res.stderr}, status=500)
524
525        # Save feedback record
526        fb = Feedback.objects.create(
527            unique_id=uid, user_id=data['user_id'],
528            feedback_text=data['feedback_text'],
529            video_url=rel,
530            trimmed_video=f"feedbackvideos/{out_file.name}",
531            start_time=data['start_time'],
532            end_time=data['end_time']
533        )
534        return JsonResponse({'status': 'success', 'id': uid, 'trimmed_video': fb.trimmed_video})
535
536    except json.JSONDecodeError:
537        return JsonResponse({'error': 'Invalid JSON'}, status=400)
538    except Exception as e:
