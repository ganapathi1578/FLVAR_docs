 26class PullAndStartAPI(APIView):
 27    """
 28    Pulls a modelâ€™s base Docker image tarball, loads it, tags it, and starts a container.
 29
 30    POST data:
 31        {
 32            "model_id": "<model identifier>"
 33        }
 34
 35    Returns:
 36        201: serialized ModelContainer with status and timestamps.
 37        400: if model_id missing (ParseError).
 38        500: on unexpected errors.
 39    """
 40    permission_classes = [AllowAny]
 41
 42    def post(self, request):
 43        model_id = request.data.get('model_id')
 44        if not model_id:
 45            raise ParseError("Provide model_id")
 46
 47        # Download, load, tag, and start container
 48        tar = download_image_tar(model_id)
 49        tag = DOCKER.load_and_tag(tar, model_id, 'base')
 50        cont = DOCKER.run(tag, model_id)
 51
 52        # Persist or update container record
 53        obj, _ = ModelContainer.objects.update_or_create(
 54            model_id=model_id,
 55            defaults={
 56                'version_id': 'base',
 57                'container_name': model_id,
 58                'status': cont.status
 59            }
 60        )
 61        return Response(ModelContainerSerializer(obj).data, status=201)

152class SwitchVersionAPI(APIView):
153    """
154    Switches a container to a different model version:
155     - Pulls weights if version not present.
156     - Instructs container to update its active version.
157
158    POST data:
159        {
160            "model_id":   "<model identifier>",
161            "version_id": "<new version identifier>"
162        }
163
164    Returns:
165        200: serialized ModelContainer with new version and status.
166        400: if missing parameters.
167        500: on unexpected errors.
168    """
169    permission_classes = [AllowAny]
170
171    def post(self, request):
172        model = request.data.get('model_id')
173        new_ver = request.data.get('version_id')
174        if not model or not new_ver:
175            raise ParseError("Provide model_id and version_id")
176
177        obj = ModelContainer.objects.filter(model_id=model).first()
178        if not obj:
179            # If container not yet pulled, pull base first
180            PullAndStartAPI().post(request)
181            obj = ModelContainer.objects.get(model_id=model)
182
183        cname = obj.container_name
184
185        # Check available versions inside the container
186        resp = requests.get(f"http://{cname}/get_versions/")
187        versions = resp.json().get('versions', [])
188
189        # Upload weights if needed
190        if new_ver not in versions:
191            filename, file_content = download_version_weights(model, new_ver)
192            file_obj = BytesIO(file_content); file_obj.name = filename
193            files = {'file': (filename, file_obj)}
194            data = {'version_id': new_ver}
195            up = requests.get(f"http://{cname}/upload_version", files=files, data=data)
196            up.raise_for_status()
197
198        # Instruct the container to switch version
199        sw = requests.get(f"http://{cname}/updateversion", json={'version_id': new_ver})
200        sw.raise_for_status()
201
202        obj.version_id = new_ver
203        obj.save()
204        return Response(ModelContainerSerializer(obj).data)

228class ModelListByIdsAPI(APIView):
229    """
230    Fetches marketplace entries for models you own, merging with local status.
231
232    POST data:
233        None in request body.
234
235    Process:
236     1. Collects all local ModelContainer.model_id values.
237     2. Calls central /filter/ API with owned IDs.
238     3. Merges each returned modelâ€™s data with local `status` and `created_at`.
239
240    Returns:
241        200: {"models": [ ...merged model objects... ]}
242        500: on errors contacting central or config issues.
243    """
244    permission_classes = [AllowAny]
245
246    def post(self, request):
247        try:
248            containers = ModelContainer.objects.all()
249            owned_ids = [c.model_id for c in containers]
250            if not owned_ids:
251                return Response({'models': []}, status=200)
252
253            conf = config.load()
254            house_id = conf.get("house_id")
255            secret_key = conf.get("secret_key")
256            if not house_id or not secret_key:
257                return Response({"error": "Missing house_id or secret_key in config."}, status=500)
258
259            auth_hash = hashlib.sha256((house_id + secret_key).encode()).hexdigest()
260            headers = {
261                "X-House-ID":   house_id,
262                "X-Auth-Hash":  auth_hash,
263                "Content-Type": "application/json"
264            }
265
266            url = "https://services.iittp.ac.in/var/vendors/models/filter/"
267            response = requests.post(url, headers=headers, json={"model_ids": owned_ids})
268
269            if response.status_code != 200:
270                return Response(
271                    {"error": "Central server error", "detail": response.text},
272                    status=response.status_code
273                )
274
275            container_map = {c.model_id: c for c in containers}
276            marketplace_models = response.json().get("models", [])
277
278            result = []
279            for model in marketplace_models:
280                mid = model.get("model_id")
281                if mid in container_map:
282                    cont = container_map[mid]
283                    model["status"] = cont.status
284                    model["created_at"] = cont.created_at.isoformat()
285                    result.append(model)
286
287            return Response({"models": result}, status=200)
288
289        except Exception as e:
290            return Response({"error": str(e)}, status=500)

295class FilteredModelDetailWithContainerInfoAPI(APIView):
296    """
297    Retrieves detailed model info from central marketplace, filters to versions
298    present in the local container, and annotates with the current deployed version.
299
300    POST data:
301        {
302            "model_id": "<model identifier>"
303        }
304
305    Returns:
306        200: filtered model data with 'versions' list and 'current_version'.
307        400: if model_id missing.
308        500: on central/container errors.
309    """
310    permission_classes = [AllowAny]
311
312    def post(self, request):
313        model_id = request.data.get("model_id")
314        if not model_id:
315            return Response({"error": "Missing 'model_id' in request body."}, status=400)
316
317        try:
318            # Load credentials
319            conf = config.load()
320            house_id = conf.get("house_id")
321            secret_key = conf.get("secret_key")
322            if not house_id or not secret_key:
323                return Response({"error": "Missing house_id or secret_key in config."}, status=500)
324
325            auth_hash = hashlib.sha256((house_id + secret_key).encode()).hexdigest()
326            headers = {
327                "X-House-ID":   house_id,
328                "X-Auth-Hash":  auth_hash,
329                "Content-Type": "application/json"
330            }
331
332            # Fetch full metadata from central
333            meta_url = "https://services.iittp.ac.in/var/vendors/models/model_detail/"
334            model_resp = requests.get(meta_url, headers=headers, json={"model_id": model_id}, timeout=5)
335            model_resp.raise_for_status()
336            model_data = model_resp.json()
337
338            # Query container for available versions
339            try:
340                cont_versions = requests.get(f"http://{model_id}/version/", timeout=5)
341                cont_versions.raise_for_status()
342                available_versions = cont_versions.json()
343            except requests.RequestException:
344                available_versions = []
345
346            # Get current deployed version
347            try:
348                container = ModelContainer.objects.get(model_id=model_id)
349                current_version = container.version_id
350            except ModelContainer.DoesNotExist:
351                current_version = None
352
353            # Filter versions
354            filtered_versions = [
355                v for v in model_data.get("versions", [])
356                if v["version_id"] in available_versions
357            ]
358
359            model_data["versions"] = filtered_versions
360            model_data["current_version"] = current_version
361
362            return Response(model_data, status=200)
363
364        except requests.RequestException as e:
365            return Response(
366                {"error": f"Marketplace/container unreachable: {str(e)}"},
367                status=500
368            )
369        except Exception as e:
370            return Response({"error": str(e)}, status=500)

317
318        try:
319            # Load credentials
320            conf = config.load()
321            house_id = conf.get("house_id")
322            secret_key = conf.get("secret_key")

352                current_version = None
353
354            # Filter versions
355            filtered_versions = [
356                v for v in model_data.get("versions", [])
357                if v["version_id"] in available_versions
358            ]
